public static void CON()
{
    try
    {
        if (CN.State == ConnectionState.Open)
            CN.Close();
        CN = new SQLiteConnection(@"Data Source=C:\dbase\DB.sqlite; Version=3; Pooling=True; Timeout=50;");
        CN.Open();
    }
    catch (Exception ex)
    {
        MessageBox.Show("Connection Error: " + ex.Message);
    }
}
